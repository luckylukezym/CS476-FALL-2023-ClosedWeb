// create users table
create table users(
    id int AUTO_INCREMENT,
    username varchar(100),
    password varchar(100),
    email varchar(100),
    PRIMARY key (id)
);


// create product table
CREATE TABLE products(
  id int auto_increment,
  name varchar(100),
  quantity int,
  price int,
  shipping int,
  image_url varchar(100),
  description text,
  bone text,
  btwo text,
  bthree text,
  primary key(id)
);
